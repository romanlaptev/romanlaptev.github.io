	var COFFEE_MACHINE_CAPACITY = 500;
	var WATER_TEMPERATURE = 80;

	var self = this;
	var timer;

	function getBoilTime()
	{
		return self.waterAmount * WATER_HEAT_CAPACITY * WATER_TEMPERATURE;
	}

	function onReady()
	{
		alert("Coffee is ready!");
	}

	this.run = function(){
		timer = setTimeout( onReady, getBoilTime() );
	}

		}
